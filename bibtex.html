To reference an individual chapter:

@InCollection{AuthorLastName12,
  author = {Author(s) Name},
  title = {Chapter Title},
  booktitle = {{O}pen{GL} {I}nsights},
  pages = {First Page-Last Page},
  editor = {Patrick Cozzi and Christophe Riccio},
  month = {July},
  year = {2012},
  isbn = {978-1439893760},
  publisher = {CRC Press},
  note = {\url{http://www.openglinsights.com/}}
}
